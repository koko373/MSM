  <h1>Savings Calculator</h1>

  <form>

    <label for="income">Monthly Income:</label><br>

    <input type="number" id="income" name="income"><br>

    <label for="rent">Rent:</label><br>

    <input type="number" id="rent" name="rent"><br>

    <label for="gas">Gas Money:</label><br>

    <input type="number" id="gas" name="gas"><br>

    <label for="bills">Bills:</label><br>

    <input type="number" id="bills" name="bills"><br>

    <label for="food">Food:</label><br>

    <input type="number" id="food" name="food"><br>

    <button onclick="calculateSavings()">Calculate Savings</button>

  </form> 

  <p id="savings"></p>

  

  <script>

    function calculateSavings() {

      // Get the input values

      var income = document.getElementById("income").value;

      var rent = document.getElementById("rent").value;

      var gas = document.getElementById("gas").value;

      var bills = document.getElementById("bills").value;

      var food = document.getElementById("food").value;

      

      // Calculate the total monthly spending

      var totalSpending = parseInt(rent) + parseInt(gas) + parseInt(bills) + parseInt(food);

      

      // Calculate the monthly savings

      var monthlySavings = income - totalSpending;

      

      // Calculate the annual savings

      var annualSavings = monthlySavings * 12;

      

      // Update the savings paragraph

      document.getElementById("savings").innerHTML = "You can save $" + monthlySavings + " each month and $" + annualSavings + " each year.";

    }

  </script>
